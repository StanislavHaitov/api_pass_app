# Ensure Minikube is running before running your Terraform commands manually.
minikube profile list

# Initial Deployment with Terraform:
terraform init
terraform apply

# Making Changes in YAML:
kubectl apply -f deployment.yaml

# If you want to keep Terraform in sync, update the corresponding main.tf resource to reflect the new state.

# After deployment find the Minikube IP and NodePort that is assigned to your service:
minikube -p project-app ip
kubectl get svc passapp -o jsonpath='{.spec.ports[0].nodePort}'

# Once you have the Minikube IP and the NodePort, you can access your app
http://<minikube-ip>:<node-port>/password/
# For example:
http://192.168.58.2:31606/password/
